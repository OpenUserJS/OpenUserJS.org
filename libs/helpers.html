's' : '') + ' ago'; } if (difference > day) { ret = date.getDate() + ' ' + months[date.getMonth()] + ' ' + date.getFullYear(); } else if (difference > hour) { ret = pluralize(Math.round(difference / hour), 'hour'); } else if (difference > minute) { ret = pluralize(Math.round(difference / minute), 'minute'); } else { ret = pluralize(Math.ceil(difference / second), 'second'); } return ret; }

exports.forIn = function (obj, forProp) { var key = null; for (key in obj) { if (!Object.prototype.hasOwnProperty.call(obj, key)) { continue; } forProp(obj[key], key, obj); } };
